DROP TABLE DRIVER;



CREATE TABLE DRIVER


(


   DRIVER_ID             NUMBER(6,0)   NOT NULL,


   DRIVER_NAME       VARCHAR(30)    NOT NULL,


   PHONE_NUMBER      VARCHAR(12)    NOT NULL,


   DRIVER_ADDRESS    VARCHAR(30)    NOT NULL,


   CONSTRAINT DRIVER_PK PRIMARY KEY(DRIVER_ID),


   CONSTRAINT DRIVER_ID_CK CHECK (DRIVER_ID <= 1000000)


);



DROP TABLE BUS_2;



CREATE TABLE BUS_2


(


   BUS_ID       CHAR(7)     NOT NULL,


   NUMBER_PLATE     VARCHAR2(30)    NOT NULL,


   BUS_CAPACITY     NUMBER(3,0)  NOT NULL,


   CONSTRAINT BUS_PK PRIMARY KEY(BUS_ID)


);



DROP TABLE ROUTE;



CREATE TABLE ROUTE


(


   ROUTE_NUMBER NUMBER(3,0) NOT NULL,


   DEPARTURE_PLACE VARCHAR(20) NOT NULL,


   ARRIVAL_PLACE VARCHAR(20) NOT NULL, 


   DEPARTURE_TIME        CHAR(10)  NOT NULL,


   ARRIVAL_TIME    CHAR(10)     NOT NULL,


   TIME_TAKEN     CHAR(10),


   CONSTRAINT ROUTE_PK PRIMARY KEY(ROUTE_NUMBER)


);



DROP TABLE TRIP;



CREATE TABLE TRIP


(


   TRIP_NUMBER      NUMBER(3,0) NOT NULL,


   DATE_TRIP       DATE     NOT NULL,


   BUS_ID_ASSIGN    CHAR(7) NOT NULL,


   DRIVER_ID_ASSIGN NUMBER(6,0) NOT NULL,


   CONSTRAINT TRIP_PK PRIMARY KEY(TRIP_NUMBER)


);



DROP TABLE SCHEDULE;



CREATE TABLE SCHEDULE


(


    BUS_ID          CHAR(7) NOT NULL,


    DATE_SCHEDULE           DATE    NOT NULL,


    ROUTE_NUMBER        NUMBER(3,0) NOT NULL,


    TRIP_NUMBER     NUMBER(3,0) NOT NULL


);
